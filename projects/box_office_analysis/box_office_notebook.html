#create empty set which will hold genres
genres = set()
# add unique genres to set
for i in range(len(movies_df)):
    for j in range(len(movies_df.iloc[i][1])):
        genres.add(movies_df.iloc[i][1][j])

#convert genres from a set to a list, so that it can be added to a dataframe        
genres = list(genres)

#create datframe to hold genre data
data = {'genre': genres, 
        'profitable': [0] * 19,
        'unprofitable': [0] * 19}
genres_df = pd.DataFrame(data)

#loop through movies_df and count how many movies in each genre are successful
for i in range(len(movies_df)):
    for j in range(len(movies_df.iloc[i][1])):
        idx = genres_df.index[genres_df['genre'] == movies_df.iloc[i][1][j]].tolist()
        if(movies_df.iloc[i]['profitable'] == True):
            genres_df.loc[idx, 'profitable'] += 1
        else: 
            genres_df.loc[idx, 'unprofitable'] += 1

#sort genres_df by number of movies in total in each genre
genres_df = genres_df.sort_values(by='profitable', key=genres_df['unprofitable'].add, ascending=False)
